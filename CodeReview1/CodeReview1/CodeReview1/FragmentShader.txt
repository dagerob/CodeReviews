#version 330 core
out vec4 FragColor;

uniform vec3 lightPos;
in vec3 FragPos; 
in vec3 Normal;

void main()
{
	vec3 norm = normalize(Normal);
	vec3 lightDir = normalize(lightPos - FragPos); 
	float diff = max(dot(norm, lightDir), 0.0);
	vec3 diffuse = diff * vec3(1.0f, 0.5f, 0.0f);
	FragColor = vec4(diffuse, 1.0) + vec4(0.1f, 0.01f, 0.0f, 1.0f);
}
